**Time Complexity:** $\mathcal O(N)$.

Let's draw lines from each building denoting the minimum height of the conductor at a point needed to protect it.
Our answer will be taken from the upper envelope of those lines (a bit like a convex hull).

For convenience, let's only focus on the lines going from left to right. These lines are defined by the square-root function,
This means that the lines are concave, so any pair intersects at most once.

This means that we should use the convex-hull trick to maintain the upper envelope of these lines. Using a monotonic deque, we can implement this approach in linear time.

```cpp
#include <bits/stdc++.h>
using namespace std;

int n, h[500000], dp1[500000], dp2[500000];

double slope(int j, int k) {
	double a = (k - j) / 2.0 / (h[k] - h[j]) - (h[k] - h[j]) / 2.0;
	return a * a + k;
}

void solve(int dp[]) {
	deque<int> q;
	for (int i = 0; i < n; i++) {
		// Keep the queue monotone
		if (!i || h[q.back()] < h[i]) {
			while ((q.size() > 0 &&
					h[i] >= h[q.back()] + ceil(sqrt(i - q.back()))) ||
				   (q.size() > 1 &&
					slope(q[q.size() - 2], q.back()) > slope(q.back(), i)))
				q.pop_back();
			q.push_back(i);
		}
		while (q.size() > 1 && slope(q[0], q[1]) < i) q.pop_front();
		dp[i] = h[q[0]] + ceil(sqrt(i - q[0]));
	}
}

int main() {
	iostream::sync_with_stdio(false);
	cin.tie(0);
	cin >> n;
	for (int i = 0; i < n; i++) cin >> h[i];

	solve(dp1);
	for (int i = 0; i < n / 2; i++) swap(h[i], h[n - 1 - i]);
	solve(dp2);

	for (int i = 0; i < n; i++)
		cout << max(dp1[i], dp2[n - 1 - i]) - h[n - 1 - i] << '\n';
	return 0;
}
```

Alternatively, we can use binary search in conjunction with CHT to solve the problem in $\mathcal O(N \log N)$ time.
However, the time limit is quite tight, so one must be careful with the implementation to get 100 points.

int bet(const pi& x, const pi& y) { // O(log N) check
	if (x.s >= y.s) return N;
	if (eval(x,y.f) < y.s) return y.f-1;
	assert(x.f < y.f);

		else hi = mid-1;
	}
	return lo;
}

vd solve(vi H) {
	deque<pair<pi,int>> D; // store function, last good
	auto ins = [&](pi p) {
		while (1) {
			if (!sz(D)) break;
			D.bk.s = bet(D.bk.f,p); // last x such that D.bk is better than p
			if (sz(D) > 1 && D[sz(D)-2].s >= D.bk.s) D.pop_back();
			else break;
		}
		D.pb({p,N});
	};
	vd res;
	F0R(i,sz(H)) {
		ins({i,H[i]});
		while (sz(D) && D[0].s < i) D.pop_front();
		res.pb(eval(D[0].f,i));
	}
	return res;
}
